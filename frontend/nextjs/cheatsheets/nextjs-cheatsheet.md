# Next.js Cheatsheet

## Quick Reference Table

| Action | Code / Config | Description |
|--------|---------------|-------------|
| Create project | `npx create-next-app@latest my-app` | Scaffolds a new Next.js project with App Router by default |
| Dev server | `npm run dev` | Starts development server with hot module replacement (default :3000) |
| Production build | `npm run build` | Creates an optimized production build with static/SSR pages |
| Start production | `npm start` | Starts the production server after `npm run build` |
| Static export | `next.config.js` → `output: 'export'` | Generates a fully static site (no Node.js server needed) |
| Page route | `app/page.tsx` | Maps to `/` (root route) |
| Dynamic route | `app/blog/[slug]/page.tsx` | Maps to `/blog/:slug` |
| Catch-all route | `app/blog/[...slug]/page.tsx` | Maps to `/blog/a/b/c` → `{ slug: ['a','b','c'] }` |
| Optional catch-all | `app/blog/[[...slug]]/page.tsx` | Maps to `/blog` and `/blog/a/b` |
| Route group | `app/(marketing)/about/page.tsx` | Organizes routes without affecting the URL path |
| Parallel route | `app/@team/page.tsx` and `app/@analytics/page.tsx` | Renders multiple independent slot views on the same layout |
| Intercepting route | `app/feed/(..)photo/[id]/page.tsx` | Loads a route from within the current layout (modal patterns) |
| Loading UI | `app/blog/loading.tsx` | Shows an instant loading state (Suspense boundary) |
| Error UI | `app/blog/error.tsx` | Client-side error boundary for a route segment |
| Not found | `app/blog/not-found.tsx` | Custom 404 page for a specific route segment |
| Layout | `app/blog/layout.tsx` | Shared UI that persists across child routes (does not re-render) |
| Template | `app/blog/template.tsx` | Shared UI that re-renders on every navigation (re-mounts state) |
| Middleware | `middleware.ts` at project root | Runs code before every request (redirects, rewrites, headers) |

## Common Commands

### Project Scaffolding

```bash
# Create a new Next.js 14+ project with App Router (default)
npx create-next-app@latest my-app --typescript --tailwind --eslint

# With specific options
npx create-next-app@latest my-app --typescript --app --src-dir --import-alias "@/*"

# Upgrade Next.js in an existing project
npm install next@latest react@latest react-dom@latest
```

### Development

```bash
# Start development server
npm run dev

# Build for production
npm run build

# Analyze bundle size (requires @next/bundle-analyzer)
ANALYZE=true npm run build

# Start production server
npm start

# Lint all files
npm run lint

# Run lint with auto-fix
npm run lint -- --fix
```

### Environment Variables

```bash
# Development only
# .env.local — overrides all other .env files
DATABASE_URL="postgresql://user:pass@localhost:5432/db"

# All environments (committed to repo)
# .env
NEXT_PUBLIC_API_URL="https://api.example.com"

# Production only
# .env.production
API_SECRET="production-secret"
```

### Static Exports

```bash
# Configure in next.config.js
# output: 'export'
# Then build and deploy the 'out/' folder to any static host
npm run build

# Serve locally for testing
npx serve@latest out
```

### Data Fetching — Server Component

```typescript
// app/products/page.tsx
async function getProducts() {
  const res = await fetch('https://api.example.com/products', {
    next: { revalidate: 3600 }, // ISR: revalidate every hour
  });
  if (!res.ok) throw new Error('Failed to fetch products');
  return res.json();
}

export default async function ProductsPage() {
  const products = await getProducts();

  return (
    <ul>
      {products.map((product: any) => (
        <li key={product.id}>{product.name}</li>
      ))}
    </ul>
  );
}
```

### Data Fetching — Client Component with SWR

```typescript
// app/dashboard/page.tsx
'use client';

import useSWR from 'swr';

const fetcher = (url: string) => fetch(url).then((r) => r.json());

export default function Dashboard() {
  const { data, error, isLoading } = useSWR('/api/user', fetcher, {
    refreshInterval: 30000, // poll every 30 seconds
  });

  if (isLoading) return <p>Loading...</p>;
  if (error) return <p>Failed to load user data</p>;

  return <pre>{JSON.stringify(data, null, 2)}</pre>;
}
```

### Middleware

```typescript
// middleware.ts
import { NextResponse } from 'next/server';
import type { NextRequest } from 'next/server';

export function middleware(request: NextRequest) {
  const token = request.cookies.get('session')?.value;
  const { pathname } = request.nextUrl;

  // Redirect unauthenticated users to login
  if (!token && pathname.startsWith('/dashboard')) {
    return NextResponse.redirect(new URL('/login', request.url));
  }

  // Set custom headers
  const response = NextResponse.next();
  response.headers.set('x-request-id', crypto.randomUUID());
  return response;
}

export const config = {
  matcher: ['/dashboard/:path*', '/api/protected/:path*'],
};
```

### API Route (Route Handler)

```typescript
// app/api/products/route.ts
import { NextResponse } from 'next/server';

export async function GET(request: Request) {
  const { searchParams } = new URL(request.url);
  const category = searchParams.get('category');

  const products = await db.products.findMany({
    where: category ? { category } : undefined,
  });

  return NextResponse.json(products);
}

export async function POST(request: Request) {
  const body = await request.json();
  const product = await db.products.create({ data: body });

  return NextResponse.json(product, { status: 201 });
}
```
